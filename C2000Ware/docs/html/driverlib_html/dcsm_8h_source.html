259 //*****************************************************************************
260 static inline DCSM_SecurityStatus
261 DCSM_getZone1CSMSecurityStatus(void)
262 {
263  uint16_t status;
264  DCSM_SecurityStatus returnStatus;
265  status = HWREGH(DCSM_Z1_BASE + DCSM_O_Z1_CR);
266 
267  //
268  // if ARMED bit is set and UNSECURED bit or ALLONE bit or both UNSECURED
269  // and ALLONE bits are set then CSM is unsecured. Else it is secure.
270  //
271  if(((status & DCSM_Z1_CR_ARMED) != 0U) &&
272  (((status & DCSM_Z1_CR_UNSECURE) != 0U) ||
273  ((status & DCSM_Z1_CR_ALLONE) != 0U )))
274  {
275  returnStatus = DCSM_STATUS_UNSECURE;
276  }
277  else if((status & DCSM_Z1_CR_ALLZERO) == DCSM_Z1_CR_ALLZERO)
278  {
279  returnStatus = DCSM_STATUS_LOCKED;
280  }
281  else
282  {
283  returnStatus = DCSM_STATUS_SECURE;
284  }
285 
286  return(returnStatus);
287 }
