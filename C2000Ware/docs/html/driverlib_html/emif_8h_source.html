404 EMIF_setAccessProtection(uint32_t configBase, uint16_t access)
405 {
406  uint16_t temp;
407  //
408  // Check the arguments.
409  //
410 
411  ASSERT(EMIF_isEMIF1ConfigBaseValid(configBase) ||
412  EMIF_isEMIF2ConfigBaseValid(configBase));
413  if(configBase == EMIF1CONFIG_BASE)
414  {
415  ASSERT(access <= EMIF_ACCPROT0_MASK_EMIF1);
416  temp = EMIF_ACCPROT0_MASK_EMIF1;
417  }
418  else
419  {
420  ASSERT(access <= EMIF_ACCPROT0_MASK_EMIF2);
421  temp = EMIF_ACCPROT0_MASK_EMIF2;
422  }
423 
424  //
425  // Sets the bits that enables access protection config.
426  //
427  EALLOW;
428  HWREGH(configBase + MEMCFG_O_EMIF1ACCPROT0) =
429  (HWREGH(configBase + MEMCFG_O_EMIF1ACCPROT0) & ~(temp)) | access;
430  EDIS;
431 }
