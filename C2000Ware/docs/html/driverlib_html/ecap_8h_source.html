205 static inline void ECAP_setEventPrescaler(uint32_t base,
206  uint16_t preScalerValue)
207 {
208  ASSERT(ECAP_isBaseValid(base));
209 
210  ASSERT(preScalerValue < ECAP_MAX_PRESCALER_VALUE);
211 
212  // write to PRESCALE bit
213  HWREGH(base + ECAP_O_ECCTL1) =
214  ((HWREGH(base + ECAP_O_ECCTL1) & (~ECAP_ECCTL1_PRESCALE_M)) |
215  (preScalerValue << ECAP_ECCTL1_PRESCALE_S));
216 }

236 static inline void ECAP_setEventPolarity(uint32_t base,
237  ECAP_Events event,
238  ECAP_EventPolarity polarity)
239 {
240 
241  uint16_t shift;
242 
243  ASSERT(ECAP_isBaseValid(base));
244 
245  shift = ((uint16_t)event) << 1U;
246 
247  // write to CAP1POL, CAP2POL, CAP3POL or CAP4POL
248  HWREGH(base + ECAP_O_ECCTL1) =
249  (HWREGH(base + ECAP_O_ECCTL1) & ~(1U << shift)) |
250  ((uint16_t)polarity << shift);
251 }

937 static inline uint32_t ECAP_getEventTimeStamp(uint32_t base, ECAP_Events event)
938 {
939  uint32_t count;
940 
941  ASSERT(ECAP_isBaseValid(base));
942 
943 
944  switch(event)
945  {
946  case ECAP_EVENT_1:
947  // read CAP1 register
948  count = HWREG(base + ECAP_O_CAP1);
949  break;
950 
951  case ECAP_EVENT_2:
952  // read CAP2 register
953  count = HWREG(base + ECAP_O_CAP2);
954  break;
955 
956  case ECAP_EVENT_3:
957  // read CAP3 register
958  count = HWREG(base + ECAP_O_CAP3);
959  break;
960 
961  case ECAP_EVENT_4:
962  // read CAP4 register
963  count = HWREG(base + ECAP_O_CAP4);
964  break;
965 
966  default:
967  // invalid event parameter
968  count = 0U;
969  break;
970  }
971 
972  return(count);
973 }
