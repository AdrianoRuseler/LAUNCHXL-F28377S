582 static inline void SDFM_enableInterrupt(uint32_t base,
583  SDFM_FilterNumber filterNumber,
584  uint16_t intFlags)
585 {
586  uint16_t offset;
587 
588  ASSERT(SDFM_isBaseValid(base));
589 
590  offset = (uint16_t)filterNumber * 16U;
591 
592  EALLOW;
593  // Low, high threshold, Modulator failure
594  if((intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
595  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
596  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT)) != 0U)
597  {
598  // Set IEL or IEH or MFIE bit of SDFM_O_SDCPARMx
599  HWREGH(base + SDFM_O_SDCPARM1 + offset) |=
600  (intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
601  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
602  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT));
603  }
604 
605  // Data filter acknowledge interrupt
606  if((intFlags & SDFM_DATA_FILTER_ACKNOWLEDGE_INTERRUPT) != 0U)
607  {
608  HWREGH(base + SDFM_O_SDDFPARM1 + offset) |= SDFM_SDDFPARM1_AE;
609  }
610  EDIS;
611 }

631 static inline void SDFM_disableInterrupt(uint32_t base,
632  SDFM_FilterNumber filterNumber,
633  uint16_t intFlags)
634 {
635  uint16_t offset;
636 
637  ASSERT(SDFM_isBaseValid(base));
638 
639  offset = (uint16_t)filterNumber * 16U;
640 
641  EALLOW;
642  // Low, high threshold, modulator failure interrupts
643  if((intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
644  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
645  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT)) != 0U)
646  {
647  // Set IEL or IEH or MFIE bit of SDFM_O_SDCPARMx
648  HWREGH(base + SDFM_O_SDCPARM1 + offset) &=
649  ~(intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
650  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
651  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT));
652  }
653 
654  // Data filter acknowledge interrupt
655  if((intFlags & SDFM_DATA_FILTER_ACKNOWLEDGE_INTERRUPT) != 0U)
656  {
657  HWREGH(base + SDFM_O_SDDFPARM1 + offset) &= ~SDFM_SDDFPARM1_AE;
658  }
659  EDIS;
660 }
