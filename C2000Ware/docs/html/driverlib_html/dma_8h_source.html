221 DMA_initController(void)
222 {
223  EALLOW;
224 
225  // Set the hard reset bit. One NOP is required after HARDRESET.
226  HWREGH(DMA_BASE + DMA_O_CTRL) |= DMA_CTRL_HARDRESET;
227  NOP;
228 
229  EDIS;
230 }
