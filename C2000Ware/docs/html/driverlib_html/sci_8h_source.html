262 static inline void
263 SCI_lockAutobaud(uint32_t base)
264 {
265  //
266  // Check the arguments.
267  //
268  ASSERT(SCI_isBaseValid(base));
269 
270  //
271  // Prime the baud register
272  //
273  HWREGH(base + SCI_O_HBAUD) = 0x0U;
274  HWREGH(base + SCI_O_LBAUD) = 0x1U;
275 
276  //
277  // Prepare for autobaud detection.
278  // Set the CDC bit to enable autobaud detection and clear the ABD bit.
279  //
280  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_CDC;
281  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_ABDCLR;
282 
283  //
284  // Wait until we correctly read an 'A' or 'a' and lock
285  //
286  while((HWREGH(base + SCI_O_FFCT) & SCI_FFCT_ABD) != SCI_FFCT_ABD)
287  {
288  }
289 
290  //
291  // After autobaud lock, clear the ABD and CDC bits
292  //
293  HWREGH(base + SCI_O_FFCT) |= SCI_FFCT_ABDCLR;
294  HWREGH(base + SCI_O_FFCT) &= ~SCI_FFCT_CDC;
295 }
