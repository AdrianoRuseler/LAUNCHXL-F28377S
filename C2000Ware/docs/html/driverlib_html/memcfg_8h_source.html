251 static inline void
252 MemCfg_setCLAMemType(uint32_t ramSections, MemCfg_CLAMemoryType claMemType)
253 {
254  //
255  // Check the arguments.
256  //
257  ASSERT((ramSections & MEMCFG_SECT_TYPE_MASK) == MEMCFG_SECT_TYPE_LS);
258 
259  //
260  // Write the CLA memory configuration to the appropriate register. Either
261  // set or clear the bit that determines the function of the RAM section as
262  // it relates to the CLA.
263  //
264  EALLOW;
265 
266  if(claMemType == MEMCFG_CLA_MEM_PROGRAM)
267  {
268  // Program memory
269  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) |= ramSections;
270  }
271  else
272  {
273  // Data memory
274  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) &= ~ramSections;
275  }
276 
277  EDIS;
278 }
