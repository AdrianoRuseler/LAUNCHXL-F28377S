421 static inline void
422 CAN_selectClockSource(uint32_t base, CAN_ClockSource source)
423 {
424  //
425  // Check the arguments.
426  //
427  ASSERT(CAN_isBaseValid(base));
428 
429  //
430  // Determine the CAN controller and set specified clock source
431  //
432  EALLOW;
433 
434  switch(base)
435  {
436  case CANA_BASE:
437  HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL2) &=
438  ~SYSCTL_CLKSRCCTL2_CANABCLKSEL_M;
439 
440  HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL2) |= ((uint16_t)source <<
441  SYSCTL_CLKSRCCTL2_CANABCLKSEL_S);
442  break;
443 
444  case CANB_BASE:
445  HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL2) &=
446  ~SYSCTL_CLKSRCCTL2_CANBBCLKSEL_M;
447 
448  HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL2) |= ((uint16_t)source <<
449  SYSCTL_CLKSRCCTL2_CANBBCLKSEL_S);
450  break;
451 
452  default:
453  // Do nothing. Not a valid mode value.
454  break;
455  }
456 
457  EDIS;
458 }
