354 static inline uint32_t
355 GPIO_readPin(uint32_t pin)
356 {
357  volatile uint32_t *gpioDataReg;
358 
359  //
360  // Check the arguments.
361  //
362  ASSERT(GPIO_isPinValid(pin));
363 
364  gpioDataReg = (uint32_t *)GPIODATA_BASE +
365  ((pin / 32U) * GPIO_DATA_REGS_STEP);
366 
367  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);
368 }
