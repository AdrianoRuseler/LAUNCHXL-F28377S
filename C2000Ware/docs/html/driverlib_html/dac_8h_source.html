436 DAC_getOffsetTrim(uint32_t base)
437 {
438  uint16_t value;
439 
440  //

443  ASSERT(DAC_isBaseValid(base));
444 
445  //
446  // Get the sign-extended offset trim value
447  //
448  value = (HWREGH(base + DAC_O_TRIM) & DAC_TRIM_OFFSET_TRIM_M);
449  value = ((value & (uint16_t)DAC_REG_BYTE_MASK) ^ (uint16_t)0x80) -
